\begin{conceptbox}
\textbf{Ideas clave} \hfill \small\textit{Holton, Cap.~8}
\medskip

Un \textbf{filtro anal\'ogico} es un sistema LTI en tiempo continuo.
Su comportamiento queda descrito por la \textbf{funci\'on de transferencia}:
$$H(s) = \frac{Y(s)}{X(s)}, \qquad s = \sigma + j\Omega$$
La \textbf{respuesta en frecuencia} se obtiene evaluando sobre el eje imaginario: $H(j\Omega)$.
Para que el sistema sea \textbf{estable}: todos los polos deben estar en el \textbf{semiplano izquierdo} ($\mathrm{Re}(s)<0$).

\smallskip
\textbf{M\'etodo gr\'afico (polos y ceros):} el m\'odulo y la fase se leen \emph{geom\'etricamente} desde el plano $s$:
$$|H(j\Omega)| = \frac{\prod \text{distancias a los ceros}}{\prod \text{distancias a los polos}},
\qquad \angle H(j\Omega) = \sum\angle\,\text{ceros} - \sum\angle\,\text{polos}$$
Cada distancia y \'angulo se miden desde el punto $s=j\Omega$ hasta cada singularidad.
\end{conceptbox}

\section{Segundo Orden: $Q$, $\omega_0$ y M\'etodo Gr\'afico}

\begin{exbox}
\textbf{Ejercicio 2.} Un filtro pasa-bajos de segundo orden tiene la funci\'on de transferencia:
$$H(s) = \frac{\omega_0^2}{s^2 + \dfrac{\omega_0}{Q}\,s + \omega_0^2}$$
con $f_0 = 1\,\text{kHz}$ y $Q = 5$.

\begin{enumerate}[label=\textbf{\alph*)}]
  \item Calcule los dos polos $s_{1,2}$ con:
  $$s_{1,2} = -\frac{\omega_0}{2Q} \pm j\,\omega_0\sqrt{1 - \frac{1}{4Q^2}}$$
  Verifique que tienen $\mathrm{Re}(s) < 0$.

  \item Grafique los polos en el plano $s$. Marque su m\'odulo $|s_k|$ y su \'angulo respecto al
  eje real negativo. ?`C\'omo se relaciona $|s_k|$ con $\omega_0$?

  \item \textbf{M\'etodo gr\'afico}: para $\Omega = 0$, $\Omega = \omega_0/2$ y $\Omega = \omega_0$,
  razone cualitativamente si $|H(j\Omega)|$ crece o cae usando la geometr\'ia de las distancias.
  No hace falta calcular, solo describir.

  \item ?`Qu\'e le pasa a la respuesta si se aumenta $Q$ (polos se acercan al eje imaginario)?
  ?`Aparece un pico? ?`Eso es deseable en un filtro pasa-bajos?

\end{enumerate}
\end{exbox}

\section{Filtro Butterworth: Polos y Orden}

\begin{conceptbox}
\textbf{Filtro Butterworth} \hfill \small\textit{Holton, Secci\'on 8.2.2}

\smallskip
Caracter\'istica: \textbf{m\'axima planicidad} en la banda de paso (sin rizado).
Su m\'odulo cuadrado es:
$$|H(j\Omega)|^2 = \frac{1}{1 + \left(\Omega/\Omega_c\right)^{2N}}$$
Los polos se distribuyen uniformemente sobre un c\'irculo de radio $\Omega_c$ en el plano $s$.
Se retienen los $N$ del \textbf{semiplano izquierdo}. Posici\'on de los polos:
$$s_k = \Omega_c\,e^{j\pi(2k+N+1)/(2N)}, \qquad k = 0,\ldots,N-1
\quad \Rightarrow \quad |s_k| = \Omega_c \text{ (siempre)}$$
\end{conceptbox}

\begin{exbox}
\textbf{Ejercicio 3.} Butterworth normalizado ($\Omega_c = 1\,\text{rad/s}$), orden $N = 2$.

\begin{enumerate}[label=\textbf{\alph*)}]
  \item Calcule $s_0$ y $s_1$. Expr\'eselos como $a + jb$.

  \item Grafique en el plano $s$. Verifique: (i)~m\'odulo~$= 1$, (ii)~semiplano izquierdo.

  \item Escriba $H(s) = 1\,/\,[(s-s_0)(s-s_1)]$, expanda el denominador y compare con
  la forma est\'andar. Identifique $\omega_0$ y $Q$.
  ?`Cu\'al es el valor de $Q$ para el Butterworth de orden~2?

  \item Usando el m\'etodo gr\'afico, explique por qu\'e $|H(j\Omega)|$ cae \emph{sin pico}
  al alejarse de $\Omega=0$. ?`Qu\'e geometr\'ia de los polos produce eso?
\end{enumerate}
\end{exbox}

\section{Dise\~no con Plantilla}

\begin{conceptbox}
\textbf{Plantilla de dise\~no pasa-bajos} \hfill \small\textit{Holton, Secci\'on 8.1}

\smallskip
Cuatro par\'ametros definen la plantilla: $f_p$ (frecuencia de paso),
$T_p = -\alpha_p$ (atenuaci\'on m\'ax.\ en banda de paso [dB, valor negativo]),
$f_s$ (frecuencia de rechazo) y $T_s = -\alpha_s$ (atenuaci\'on m\'in.\ en banda de rechazo [dB]).

\smallskip
Orden m\'inimo para Butterworth (Holton, ec.~8.10):
$$N = \left\lceil \frac{\log_{10}(L)}{\log_{10}(\xi)} \right\rceil,
\quad \xi = \frac{f_s}{f_p},\quad L = \frac{\varepsilon_s}{\varepsilon_p},\quad
\varepsilon_p = \sqrt{10^{|T_p|/10}-1},\quad \varepsilon_s = \sqrt{10^{|T_s|/10}-1}$$
\end{conceptbox}

\begin{exbox}
\textbf{Ejercicio 4.} Determine el orden m\'inimo $N$ de Butterworth para cada caso.
Calcule $\xi$, $\varepsilon_p$, $\varepsilon_s$, $L$ y aplique la f\'ormula.

\medskip
\textbf{a)}\quad $f_p = 1\,\text{kHz}$, $T_p = -3\,\text{dB}$,\quad$f_s = 2\,\text{kHz}$, $T_s = -40\,\text{dB}$.

\medskip
\textbf{b)}\quad $f_p = 1\,\text{kHz}$, $T_p = -3\,\text{dB}$,\quad$f_s = 3\,\text{kHz}$, $T_s = -40\,\text{dB}$.

\medskip
\textbf{c)}\quad $f_p = 1\,\text{kHz}$, $T_p = -1\,\text{dB}$,\quad$f_s = 2\,\text{kHz}$, $T_s = -60\,\text{dB}$.

\medskip
Luego responda: \textbf{(i)}~?`Cu\'al requiere mayor orden? ?`Por qu\'e?
\textbf{(ii)}~Al pasar de~\textbf{a)} a~\textbf{b)}, se ampli\'o la banda de transici\'on. ?`C\'omo afect\'o a $N$?
\end{exbox}

\section{Ejercicio Integrador: Filtro Anti-Replicado + ADC}

\begin{hintbox}
\textbf{Contexto:} un filtro \emph{anti-aliasing} (o anti-replicado) es un filtro pasa-bajos
anal\'ogico que se coloca \emph{antes} del ADC para garantizar que no lleguen componentes
por encima de $f_s/2$ (Nyquist). Elegir bien su orden $N$ y su $f_c$ es cr\'itico:
si el filtro aten\'ua poco, el ADC recibe se\~nal solapada; si aten\'ua demasiado pronto,
se pierde parte de la banda \'util. Adem\'as, el n\'umero de bits del ADC pone un l\'imite
pr\'actico a la SNR y por tanto a la atenuaci\'on \emph{m\'inima} que se necesita fuera de banda.
\end{hintbox}

\begin{exbox}
\textbf{Ejercicio 5. Problema integrador.}
Se desea digitalizar una se\~nal de sensor de temperatura con las siguientes especificaciones:
\begin{itemize}[noitemsep]
  \item[\textbf{i.}] Se\~nal con ancho de banda m\'aximo de $500\,\text{Hz}$.
  \item[\textbf{ii.}] Se\~nal a partir de $f = 2\,\text{kHz}$ con atenuaci\'on m\'inima de $24\,\text{dB}$.
  \item[\textbf{iii.}] El filtro anti-replicado debe tener m\'axima planicidad en la banda de paso.
  \item[\textbf{iv.}] ADC de 8~bits, rango de entrada $-5\,\text{V}$ a $+5\,\text{V}$.
\end{itemize}

\begin{enumerate}[label=\textbf{(\alph*)}]
  \item \textbf{Orden m\'inimo del filtro.}
  Identifique qu\'e tipo de filtro corresponde al requisito de ``m\'axima planicidad (item~c).
  Luego, usando $f_p = 500\,\text{Hz}$ y $T_p = -3\,\text{dB}$ como borde de banda de paso,
  y $f_s = 2\,\text{kHz}$, $T_s = -24\,\text{dB}$, calcule el orden m\'inimo $N$.

  \item \textbf{Frecuencia de corte $f_c$.}
  Una vez obtenido $N$, calcule la frecuencia de corte $\Omega_c$ que hace que $|H(j\Omega_p)|$
  sea exactamente $T_p = -3\,\text{dB}$ (Holton, ec.~8.11):
  $$\Omega_c = \Omega_p \cdot \varepsilon_p^{-1/N}$$
  Expr\'ese $f_c$ en Hz. ?`Es $f_c$ mayor o menor que $f_p = 500\,\text{Hz}$? ?`Por qu\'e?

  \item \textbf{Frecuencia de muestreo m\'inima $f_s$.}
  El teorema de Nyquist pide $f_s > 2 f_{\max}$, pero en la pr\'actica se deja margen.
  Proponga una frecuencia de muestreo razonable $f_s$ sabiendo que la banda \'util llega
  hasta $500\,\text{Hz}$ y que el filtro ya aten\'ua $24\,\text{dB}$ a partir de $2\,\text{kHz}$.
  Justifique su elecci\'on.

  \item \textbf{Paso de cuantizaci\'on del ADC.}
  Para el ADC de 8~bits con rango $V_{FS} = 10\,\text{V}$ ($-5$ a $+5\,\text{V}$),
  calcule el paso de cuantizaci\'on $\Delta V = V_{FS}/2^N$ y la SNR te\'orica
  (Holton, ec.~8.24): $\mathrm{SNR} \approx 6{,}02 \cdot N_{\text{bits}} + 1{,}76\,\text{dB}$.
  ?`Tiene sentido usar un filtro con $24\,\text{dB}$ de atenuaci\'on m\'inima para este ADC?
  Hint: compare $T_s = -24\,\text{dB}$ con la SNR calculada.
\end{enumerate}
\end{exbox}
